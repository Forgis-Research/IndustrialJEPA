% ============================================================

    minimum height=1.3cm, minimum width=1.6cm, inner sep=0pt] (raw) {};

        ([xshift={\x cm - 0.1cm}]raw.south west)

\node[annot,below=0.15cm of raw] {$\mathbf{X}\in\mathbb{R}^{T\times S}$};
\node[font=\sffamily\scriptsize,csecondary,rotate=90,anchor=south]
    at ([xshift=-1.0cm]raw.center) {sensors};
\node[font=\sffamily\scriptsize,csecondary,anchor=north]
    at ([yshift=0.15cm]raw.south) {time $\to$};

% ============================================================
% STAGE 2: Normalize
% ============================================================

\node[stage, draw=caccent!70!black, fill=caccentbg,

    \textbf{Normalize}\\[1pt]

    {\scriptsize min-max / z-score}
};

};

% ============================================================
% STAGE 3: Tokenize
% ============================================================

\node[stage, draw=cprimary!70!black, fill=cprimarybg,
      right=0.8cm of norm] (token) {
    \textbf{Tokenize}\\[1pt]
    {\scriptsize Linear projection}\\[-1pt]
    {\scriptsize $\mathbf{x}^{(t)} \!\to\! \mathbf{e}_t$}
};
\draw[dataarrow] (norm.east) -- (token.west);
\node[annot,below=0.15cm of token,align=center] {
    $\mathbf{e}_t = \mathbf{W}\mathbf{x}^{(t)} + \mathbf{b}$\\[-2pt]

};

% ============================================================
% STAGE 4: Positional Encoding
% ============================================================

\node[stage, draw=cborder, fill=cbgcool,
      right=0.8cm of token] (pe) {
    \textbf{Position}\\[1pt]
    {\scriptsize Sinusoidal PE}\\[-1pt]
    {\scriptsize + sensor ID emb}
};
\draw[dataarrow] (token.east) -- (pe.west);
\node[annot,below=0.15cm of pe,align=center] {

};

% ============================================================
% STAGE 5: To Encoder
% ============================================================

\node[stage, draw=csecondary!70!black, fill=csecondarybg,
