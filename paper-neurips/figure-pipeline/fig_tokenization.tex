% ============================================================
% STAGE 1: Raw Multivariate Time Series
% ============================================================

\node[section] at (0,2.0) {RAW DATA};

% Draw a small matrix representation

    \draw[cneutrallight,line width=0.3pt]
        ([xshift={\x cm - 0.15cm}]raw.south west)
        -- ([xshift={\x cm - 0.15cm},yshift=1.1cm]raw.south west);
}

\node[note,below=0.15cm of raw] {$\mathbf{X}\in\mathbb{R}^{T\times S}$};
\node[note] at ([yshift=-0.55cm]raw.south) {$T$ timesteps, $S$ sensors};

% Row/col labels
\node[font=\sffamily\scriptsize,cprimary,rotate=90,anchor=south]

\node[font=\sffamily\scriptsize,cprimary,anchor=north]
    at ([yshift=-0.05cm]raw.south) {time $\to$};

% ============================================================
% STAGE 2: Normalization
% ============================================================

\node[stage=caccent,right=1.0cm of raw] (norm) {
    \textbf{Normalize}\\[1pt]
    {\scriptsize Per-sensor}\\[-1pt]
    {\scriptsize min-max / z-score}
};
\draw[dataarrow] (raw.east) -- (norm.west);

\node[note,below=0.15cm of norm,align=center] {
    Per operating\\[-2pt]
    condition if needed
};

% ============================================================
% STAGE 3: Tokenization
% ============================================================

\node[stage=cprimary,right=1.0cm of norm] (token) {
    \textbf{Tokenize}\\[1pt]
    {\scriptsize Linear projection}\\[-1pt]
    {\scriptsize $\mathbf{x}^{(t)} \!\to\! \mathbf{e}_t$}
};
\draw[dataarrow] (norm.east) -- (token.west);

% Two options shown
\node[note,below=0.15cm of token,align=center] {
    $\mathbf{e}_t = \mathbf{W}\mathbf{x}^{(t)} + \mathbf{b}$\\[-2pt]

};

% ============================================================
% STAGE 4: Positional Encoding
% ============================================================

\node[stage=csecondary,right=1.0cm of token] (pe) {
    \textbf{Position}\\[1pt]

    {\scriptsize + sensor ID emb}
};
\draw[dataarrow] (token.east) -- (pe.west);

\node[note,below=0.15cm of pe,align=center] {

    indexed, not learned

% STAGE 5: Encoder Input
