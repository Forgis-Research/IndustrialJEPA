\begin{document}
\begin{tikzpicture}[
    >=Stealth,
    font=\sffamily\small,
    dataarrow/.style={->,line width=0.7pt,ctextdark},
    softarrow/.style={->,line width=0.5pt,ctextmid},
    sechead/.style={font=\sffamily\scriptsize\bfseries,text=ctextmid},
    annot/.style={font=\sffamily\scriptsize,text=ctextmid},
    compblock/.style={
        rounded corners=2.5pt, line width=0.5pt,
        minimum height=1.1cm, minimum width=2.2cm,
        align=center, text=ctextdark, inner sep=4pt,
    },
    capsule/.style={
        rounded corners=6pt, draw=cborder, fill=white,
        line width=0.4pt, minimum height=1.2cm, minimum width=0.45cm,
        inner sep=2pt,
    },
    probe/.style={
        rounded corners=8pt, draw=cborder, fill=cbgwarm,
        line width=0.4pt, minimum height=0.42cm, minimum width=1.4cm,
        align=center, text=ctextdark, inner sep=2pt,
    },
]

% ============================================================
% SECTION HEADERS
% ============================================================
\node[sechead] at (0,3.3) {INPUT};
\node[sechead] at (2.7,3.3) {SHARED ENCODER};

\node[sechead] at (7.8,3.3) {PREDICTION};
\node[sechead] at (10.5,3.3) {LOSSES};

% Key difference callout
\node[rounded corners=2.5pt, draw=cborder, fill=cbgcool,
    font=\sffamily\tiny, text=ctextdark, align=left, inner sep=4pt,
    anchor=south west]
    at (-0.3,3.6) {
    \textbf{SIGReg variant:} shared $\theta$ (no EMA). Collapse prevented by spectral regularization.
};

% ============================================================
% INPUTS
% ============================================================
\node[rounded corners=2pt, draw=cborder, fill=white,
    minimum height=1.5cm, minimum width=1.2cm,
    inner sep=0pt] (past) at (0,1.6) {};
\begin{scope}
    \clip ([xshift=1pt,yshift=1pt]past.south west)
        rectangle ([xshift=-1pt,yshift=-1pt]past.north east);
    \foreach \yoff/\col in {0.4/csecondary, 0.2/csecondary!70, 0.0/csecondary!50,
        -0.2/csecondary!70, -0.4/csecondary!35} {
        \draw[\col,line width=0.4pt,
              decorate,decoration={snake,amplitude=1pt,segment length=4.5pt}]
            ([xshift=3pt,yshift=\yoff cm]past.center)
            -- ([xshift=-3pt,yshift=\yoff cm]past.center);
    }
\end{scope}
\node[font=\sffamily\scriptsize,csecondary,below=0.12cm of past] {$\mathbf{x}_{1:t}$};

\node[rounded corners=2pt, draw=cborder, fill=white,
    minimum height=1.5cm, minimum width=1.2cm,
    inner sep=0pt, below=0.7cm of past] (future) {};
\begin{scope}
    \clip ([xshift=1pt,yshift=1pt]future.south west)
        rectangle ([xshift=-1pt,yshift=-1pt]future.north east);
    \foreach \yoff/\col in {0.4/csecondary, 0.2/csecondary!70, 0.0/csecondary!50,
        -0.2/csecondary!70, -0.4/csecondary!35} {
        \draw[\col,line width=0.4pt,
              decorate,decoration={snake,amplitude=1pt,segment length=4.5pt}]
            ([xshift=3pt,yshift=\yoff cm]future.center)
            -- ([xshift=-3pt,yshift=\yoff cm]future.center);
    }
\end{scope}
\node[font=\sffamily\scriptsize,csecondary,below=0.12cm of future] {$\mathbf{x}_{t\!+\!1:t\!+\!k}$};

% ============================================================
% SHARED ENCODER (both blue -- same weights)
% ============================================================
\node[compblock, draw=csecondary!40, fill=csecondarybg,
    right=0.8cm of past] (enc_ctx) {
    {\scriptsize\textbf{Encoder} $f_\theta$}\\[1pt]
    {\scriptsize Causal (context)}
};
\draw[dataarrow] (past.east) -- (enc_ctx.west);

\node[compblock, draw=csecondary!40, fill=csecondarybg,
    right=0.8cm of future] (enc_tgt) {
    {\scriptsize\textbf{Encoder} $f_\theta$}\\[1pt]
    {\scriptsize Bidirectional (target)}
};
\draw[dataarrow] (future.east) -- (enc_tgt.west);

% Shared weights connection
\draw[<->,line width=0.5pt,csecondary,densely dotted]
    ([xshift=0.3cm]enc_ctx.south) -- ([xshift=0.3cm]enc_tgt.north)
    node[midway,right=0.12cm,font=\sffamily\tiny,text=csecondary] {shared $\theta$};

% ============================================================
% REPRESENTATIONS
% ============================================================
\node[capsule,right=0.55cm of enc_ctx] (h_past) {};
\foreach \y in {0.35,0.15,-0.05,-0.25,-0.45} {
    \fill[csecondary!50] ([yshift=\y cm]h_past.center) circle (1.1pt);
}
\node[annot,above=0.12cm of h_past] {$\mathbf{h}_\text{past}$};
\draw[dataarrow] (enc_ctx.east) -- (h_past.west);

\node[capsule,right=0.55cm of enc_tgt] (h_fut) {};
\foreach \y in {0.35,0.15,-0.05,-0.25,-0.45} {
    \fill[csecondary!50] ([yshift=\y cm]h_fut.center) circle (1.1pt);
}
\node[annot,above=0.12cm of h_fut] {$\mathbf{h}_\text{fut}$};
\draw[dataarrow] (enc_tgt.east) -- (h_fut.west);

% ============================================================
% PREDICTOR
% ============================================================
\node[diamond, draw=cprimary!40, fill=cprimarybg,
    line width=0.4pt, inner sep=1.5pt,
    above right=0.3cm and 1.0cm of h_past] (horizon) {{\tiny $k$}};
\node[annot,above=0.08cm of horizon,font=\sffamily\tiny] {$k\!\sim\!\mathcal{U}[5,30]$};

\node[compblock, draw=cprimary!40, fill=cprimarybg,
    right=1.0cm of h_past, minimum width=1.8cm] (predictor) {
    {\scriptsize\textbf{Predictor} $g_\phi$}
};
\draw[dataarrow] (h_past.east) -- (predictor.west);
\draw[softarrow] (horizon.south) -- (predictor.north);

\node[capsule,right=0.4cm of predictor] (h_hat) {};
\foreach \y in {0.35,0.15,-0.05,-0.25,-0.45} {
    \fill[cprimary!40] ([yshift=\y cm]h_hat.center) circle (1.1pt);
}
\node[annot,above=0.12cm of h_hat] {$\hat{\mathbf{h}}_\text{fut}$};
\draw[dataarrow] (predictor.east) -- (h_hat.west);

% ============================================================
% LOSSES
% ============================================================

% Prediction loss
\node[compblock, draw=chighlight!40, fill=chighlightbg,
    minimum width=1.2cm, minimum height=0.75cm]
    (loss_pred) at ($(h_hat.east) + (1.4,0.4)$) {
    {\scriptsize $\mathcal{L}_\text{pred}$}\\[0pt]
    {\tiny $\|\hat{\mathbf{h}}\!-\!\mathbf{h}\|_1$}
};
\draw[dataarrow] (h_hat.east) -- ++(0.2,0) |- (loss_pred.west);
\draw[dataarrow] (h_fut.east) -- ++(0.2,0) |- ([yshift=-0.08cm]loss_pred.west);

% SIGReg loss
\node[compblock, draw=chighlight!40, fill=chighlightbg,
    minimum width=1.2cm, minimum height=0.75cm,
    below=0.35cm of loss_pred] (loss_sig) {
    {\scriptsize $\mathcal{L}_\text{SIG}$}\\[0pt]
    {\tiny SIGReg}
};
\draw[softarrow,chighlight!50] (h_fut.east) -- ++(0.2,0) |- (loss_sig.west);

% Total loss
\node[annot,right=0.15cm of $(loss_pred.east)!0.5!(loss_sig.east)$,align=left] {
    {\tiny $\mathcal{L} = \mathcal{L}_\text{pred} + \lambda\mathcal{L}_\text{SIG}$}
};

% ============================================================
% PROBES (bottom strip)
% ============================================================
\coordinate (sep_y) at ([yshift=-0.4cm]future.south);
\draw[cborder,line width=0.3pt]
    ([xshift=-0.3cm]past.west |- sep_y) -- ([xshift=0.3cm]loss_pred.east |- sep_y);
\node[sechead,anchor=west] at ([xshift=-0.3cm,yshift=-0.2cm]past.west |- sep_y)

\node[probe] (p_rul) at ([xshift=0.3cm]past.east |- py) {
    {\tiny \textbf{RUL}}
};
\node[probe,right=0.3cm of p_rul] (p_tte) {
    {\tiny \textbf{TTE}}
};
\node[probe,right=0.3cm of p_tte] (p_fault) {
    {\tiny \textbf{Fault}}
};

    {\tiny \textbf{Anomaly}}
};

\node[annot,below=0.05cm of p_rul,font=\sffamily\tiny] {supervised};

\node[annot,below=0.05cm of p_fault,font=\sffamily\tiny] {supervised};
\node[annot,text=chighlight,below=0.05cm of p_anom,font=\sffamily\tiny] {zero-label};

\draw[softarrow] (h_past.south) -- ++(0,-0.35) -| (p_rul.north);

\draw[softarrow] (h_past.south) -- ++(0,-0.35) -| (p_fault.north);

\end{tikzpicture}
