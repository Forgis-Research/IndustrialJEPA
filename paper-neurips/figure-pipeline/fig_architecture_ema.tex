    },
]

% ============================================================
% PHASE LABELS

% ============================================================

% --- Past input ---
\node[rounded corners=2pt, draw=cborder, fill=white,
    minimum height=1.5cm, minimum width=1.2cm,
    inner sep=0pt] (past) at (0,1.5) {};
\begin{scope}
    \clip ([xshift=1pt,yshift=1pt]past.south west)
        rectangle ([xshift=-1pt,yshift=-1pt]past.north east);
    \foreach \yoff/\col in {0.4/csecondary,0.2/csecondary!70,

        \draw[\col,line width=0.4pt,
              decorate,decoration={snake,amplitude=1pt,segment length=4.5pt}]
            ([xshift=3pt,yshift=\yoff cm]past.center)
            -- ([xshift=-3pt,yshift=\yoff cm]past.center);
    }
\end{scope}
\node[font=\sffamily\scriptsize,csecondary,below=0.12cm of past] {$\mathbf{x}_{1:t}$};

% --- Future input ---
\node[rounded corners=2pt, draw=cborder, fill=white,
    minimum height=1.5cm, minimum width=1.2cm,
    inner sep=0pt, below=0.9cm of past] (future) {};
\begin{scope}
    \clip ([xshift=1pt,yshift=1pt]future.south west)
        rectangle ([xshift=-1pt,yshift=-1pt]future.north east);
    \foreach \yoff/\col in {0.4/caccent,0.2/caccent!70,
        0.0/caccent!50,-0.2/caccent!70,-0.4/caccent!35} {
        \draw[\col,line width=0.4pt,
              decorate,decoration={snake,amplitude=1pt,segment length=4.5pt}]
            ([xshift=3pt,yshift=\yoff cm]future.center)
            -- ([xshift=-3pt,yshift=\yoff cm]future.center);
    }
\end{scope}
\node[font=\sffamily\scriptsize,caccent,below=0.12cm of future] {$\mathbf{x}_{t\!+\!1:t\!+\!k}$};

% --- Context Encoder ---
\node[compblock, draw=csecondary!40, fill=csecondarybg,
    right=0.9cm of past] (enc_ctx) {
    {\scriptsize\textbf{Context Enc.}}\\[1pt]
    {\scriptsize $f_\theta$\enspace Causal}
};
\draw[dataarrow] (past.east) -- (enc_ctx.west);

% --- Target Encoder ---
\node[compblock, draw=caccent!40, fill=caccentbg,
    right=0.9cm of future] (enc_tgt) {
    {\scriptsize\textbf{Target Enc.}}\\[1pt]
    {\scriptsize $\bar{f}_\xi$\enspace EMA}
};
\draw[dataarrow] (future.east) -- (enc_tgt.west);

% --- EMA connection ---
\draw[->,line width=0.6pt,caccent!60,dashed]
    ([xshift=0.3cm]enc_ctx.south) -- ([xshift=0.3cm]enc_tgt.north)
    node[midway,right=0.15cm,font=\sffamily\tiny,text=caccent] {
        $\xi\!\leftarrow\!\tau\xi\!+\!(1\!-\!\tau)\theta$
    };

% --- h_past capsule ---

}

% --- h_fut capsule ---

\foreach \y in {0.35,0.15,-0.05,-0.25,-0.45} {
    \fill[caccent!50] ([yshift=\y cm]h_fut.center) circle (1.1pt);
}
\node[annot,above=0.12cm of h_fut] {$\mathbf{h}_\text{fut}$};

\draw[dataarrow] (enc_tgt.east) -- (h_fut.west);

% --- Predictor ---
\node[compblock, draw=cprimary!40, fill=cprimarybg,

    {\scriptsize\textbf{Predictor}}\\[1pt]
    {\scriptsize $g_\phi$}
};
\draw[dataarrow] (h_past.east) -- (predictor.west);

% --- Horizon k ---
\node[diamond, draw=cprimary!40, fill=cprimarybg,

\node[annot,left=0.12cm of horizon,font=\sffamily\tiny] {$k\!\sim\!\mathcal{U}[5,30]$};

\node[capsule,right=0.4cm of predictor] (h_hat) {};
\foreach \y in {0.35,0.15,-0.05,-0.25,-0.45} {
    \fill[cprimary!40] ([yshift=\y cm]h_hat.center) circle (1.1pt);
}
\node[annot,above=0.12cm of h_hat] {$\hat{\mathbf{h}}_\text{fut}$};
\draw[dataarrow] (predictor.east) -- (h_hat.west);

\draw[dataarrow] (h_fut.north) -- (loss.south);

% ============================================================
% RIGHT: Downstream Probes
% ============================================================

% Arrow from h_past to probes

\node[probe,anchor=west] (p_tte) at (10.2,1.45) {
