%           monotonic event probability surfaces via discrete hazard CDF."

% Layout: Two rows separated by dashed line. Left-to-right flow.

% Size: NeurIPS single column (5.5in / ~14cm)

  },

    text=frozenGray!60!black,
  },
  emb/.style={
    draw=#1!70!black, fill=#1!18, circle,

\begin{document}
\begin{tikzpicture}[node distance=4.5mm and 5.5mm]

% =====================================================================
%  PRETRAINING — CONTEXT PATH
% =====================================================================

\node[sechd] (pt_lbl) at (0, 0) {Pretraining};

% Input
\node[inp, below=3mm of pt_lbl.south west, anchor=north west]
  (xctx) {$\mathbf{x}_{\leq t}$};

% Tokenizer (simplified)
\node[smallblk=okBlue, right=5mm of xctx] (tok_ctx) {Tokenize};

% Causal Transformer
\node[blk=okBlue, right=5mm of tok_ctx, minimum width=1.8cm]
  (enc) {Causal Transformer};

% h_t
\node[emb=okBlue, right=5mm of enc] (ht) {$\mathbf{h}_t$};

% Predictor
\node[blk=okOrange, right=6mm of ht, minimum width=1.3cm]
  (pred) {Predictor $g_\phi$};

% Delta-t input
\node[inp, above=3mm of pred, minimum width=0.6cm, minimum height=0.38cm,
      draw=okOrange!50, fill=okOrange!5]
  (dt_in) {$\Delta t$};

% h-hat
\node[emb=okOrange, right=5mm of pred] (hhat) {$\hat{\mathbf{h}}$};

% Loss
\node[lossn, right=5mm of hhat] (loss) {$\mathcal{L}_1$};

% Context path arrows
\draw[arr] (xctx) -- (tok_ctx);
\draw[arr] (tok_ctx) -- (enc);
\draw[arr] (enc) -- (ht);
\draw[arr] (ht) -- (pred);
\draw[arr] (dt_in) -- (pred);
\draw[arr] (pred) -- (hhat);
\draw[arr] (hhat) -- (loss);

% Encoder label
\node[lbl, text=okBlue!80!black, above=1mm of enc.north] {$f_\theta$};

% =====================================================================
%  PRETRAINING — TARGET PATH
% =====================================================================

% Target input (below context input)
\node[inp, below=13mm of xctx]
  (xtgt) {$\mathbf{x}_{(t,t{+}\Delta t]}$};

% Tokenizer (simplified)
\node[smallblk=okGreen, right=5mm of xtgt] (tok_tgt) {Tokenize};

% Bidirectional Transformer
\node[blk=okGreen, right=5mm of tok_tgt, minimum width=1.8cm]
  (tgt_enc) {Bidir.\ Transformer};

% Attention pool
\node[blk=okGreen, right=5mm of tgt_enc, minimum width=0.9cm]
  (apool) {Attn Pool};

% h*
\node[emb=okGreen, right=5mm of apool] (hstar) {$\mathbf{h}^*$};

% Target path arrows
\draw[arr] (xtgt) -- (tok_tgt);
\draw[arr] (tok_tgt) -- (tgt_enc);
\draw[arr] (tgt_enc) -- (apool);
\draw[arr] (apool) -- (hstar);

% Target encoder label
\node[lbl, text=okGreen!80!black, above=1mm of tgt_enc.north] {$\bar{f}_\xi$};

% h* to loss: route upward
\draw[arr] (hstar.north) -- ++(0, 0.25)
  -| (loss.south);
\node[lbls, inner sep=1pt] at ($(loss.south) + (-0.35, -0.22)$) {sg};

% --- EMA arrow (encoder -> target encoder) ---
\draw[arrd, okGreen!70!black]
  ($(enc.south) + (-0.2, 0)$) -- ++(0, -0.25)
  -| node[lbls, pos=0.3, left=0.5pt] {EMA}
  ($(tgt_enc.north) + (-0.2, 0)$);

% --- Timeline annotation ---
\coordinate (tl_start) at ($(xctx.south west) + (0, -0.6)$);
\draw[black!40, line width=0.3pt]
  (tl_start) -- ++(2.4, 0);
\draw[okBlue!70, line width=2pt]
  (tl_start) -- ++(1.2, 0)
  node[lbls, midway, below=1.5pt, text=okBlue!80!black] {context};
\draw[okGreen!70, line width=2pt]
  ($(tl_start) + (1.2, 0)$) -- ++(1.2, 0)
  node[lbls, midway, below=1.5pt, text=okGreen!80!black] {target};
% Tick at boundary
\draw[black!50, line width=0.3pt]
  ($(tl_start) + (1.2, 0.07)$) -- ++(0, -0.14)
  node[lbls, above=1pt] {$t$};

% =====================================================================
%  DASHED SEPARATOR
% =====================================================================

\coordinate (sep_y) at ($(xtgt.south) + (0, -0.9)$);
\draw[draw=black!20, dashed, line width=0.35pt]
  (pt_lbl.west |- sep_y) -- (loss.east |- sep_y);

% =====================================================================
%  FINETUNING (pred-FT)
% =====================================================================

\node[sechd, anchor=west] (ft_lbl)
  at (pt_lbl.west |- sep_y) [yshift=-4.5mm] {Finetuning};

% Input
\node[inp, below=3mm of ft_lbl.south west, anchor=north west]
  (ft_x) {$\mathbf{x}_{\leq t}$};

% Frozen encoder (single compressed block — no internal detail)
\node[frozen, right=5mm of ft_x, minimum width=2.6cm, minimum height=0.6cm]
  (ft_enc) {Encoder $f_\theta$};

% Frozen label
\node[lbls, text=frozenGray, above=0.5mm of ft_enc.north, anchor=south]
  {\textbf{frozen}};

% h_t
\node[emb=okBlue, right=5mm of ft_enc] (ft_ht) {$\mathbf{h}_t$};

% Predictor (trainable)
\node[blk=okOrange, right=5mm of ft_ht, minimum width=1.3cm]
  (ft_pred) {Predictor $g_\phi$};

% Horizons input
\node[inp, above=3mm of ft_pred, minimum width=1.1cm, minimum height=0.38cm,
      draw=okOrange!50, fill=okOrange!5]
  (ft_dt) {$\Delta t_1\!,\ldots,\!\Delta t_K$};

% h-hat at each horizon
\node[emb=okOrange, right=5mm of ft_pred] (ft_hk) {$\hat{\mathbf{h}}_k$};

% Event head
\node[blk=okVermillion, right=5mm of ft_hk, minimum width=1.1cm]
  (ehead) {Event Head};

% Sigma
\node[smallblk=okVermillion, right=4mm of ehead, minimum height=0.45cm]
  (sig) {$\sigma$};

% Lambda
\node[emb=okVermillion, right=4mm of sig, minimum size=0.4cm]
  (lam) {$\lambda_k$};

% Survival CDF
\node[blk=okVermillion, right=4mm of lam, minimum width=1.3cm]
  (cdf) {$\displaystyle\prod_{j\leq k}\!(1{-}\lambda_j)$};

% Output
\node[emb=okVermillion, right=5mm of cdf, minimum size=0.45cm]
  (pout) {$p$};

% Finetuning arrows
\draw[arr, frozenGray] (ft_x) -- (ft_enc);
\draw[arr, frozenGray] (ft_enc) -- (ft_ht);
\draw[arr] (ft_ht) -- (ft_pred);
\draw[arr] (ft_dt) -- (ft_pred);
\draw[arr] (ft_pred) -- (ft_hk);
\draw[arr] (ft_hk) -- (ehead);
\draw[arr] (ehead) -- (sig);
\draw[arr] (sig) -- (lam);
\draw[arr] (lam) -- (cdf);
\draw[arr] (cdf) -- (pout);

% Annotations
\node[lbls, below=1mm of lam.south, align=center]
  {hazard};
\node[lbls, below=1mm of pout.south] {$p(t,\!\Delta t_k)$};

% g_phi label omitted — already inside the predictor box text

% =====================================================================
%  GROUP BOXES (background)
% =====================================================================

\begin{scope}[on background layer]
  % Context encoder group
  \node[gbox=okBlue, fit=(tok_ctx)(enc), inner sep=3.5pt] {};
  % Target encoder group
  \node[gbox=okGreen, fit=(tok_tgt)(tgt_enc)(apool), inner sep=3.5pt] {};
  % Hazard-CDF chain
  \node[gbox=okVermillion, fit=(ehead)(sig)(lam)(cdf)(pout), inner sep=3.5pt] (cdf_group) {};
\end{scope}

% CDF group label
\node[lbls, text=okVermillion!70!black, above=1mm of cdf_group.north]
  {discrete hazard $\to$ survival CDF};

\end{tikzpicture}
